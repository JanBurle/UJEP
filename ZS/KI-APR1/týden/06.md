```python
def star(n: int, l: int):
  """Draw a star with n points and line length l. n must be odd."""
  if not (0 < n and n%2): # check n validity
    raise Exception('n must be positive and odd')

  a = 180 - 180/n
  for i in range(n):
    T.forward(l)
    T.right(a)

def star2(n: int, l: int):
  """Draw a star with n points, hollow middle, and line length l."""
  a = 30 # 360/n # point angle
  a1 = 180 - a
  a2 = a - 180 + 360/n
  for i in range(n):
    T.forward(l)
    T.right(a1)
    T.forward(l)
    T.right(a2)

# do it
T.initializeTurtle(9)
T.shape('circle')
star2(11,60)
for i in range(1,2):
  skipTo(100 * i, 100)
  star(n, 80)
  skipTo(100 * i, 200)
  star2(n, 60)
```

## Nakreslete barevn√Ω rekurzivn√≠ les

```python
from random import randint
brown      = (150,75,0)   # hnƒõd√Ω kmen
darkgreen  = (89,179,0)   # tmavƒõ zelen√© vƒõtve
lightgreen = (155,255,51) # svƒõtle zelen√© list√≠

# p≈ôechod mezi barvou kmenu a vƒõtv√≠ podle hloubky
def mixColor(depth: int) -> tuple[int,int,int]:
  maxDepth = 6 # omezen√≠ hloubky pro v√Ωpoƒçet
  depth = min(depth, maxDepth)
  c2 = depth   # koeficienty barev
  c1 = 4-c2

  # p≈ôechod mezi barvami
  r1,g1,b1 = brown
  r2,g2,b2 = darkgreen
  r = int((r1*c1 + r2*c2) / maxDepth)
  g = int((g1*c1 + g2*c2) / maxDepth)
  b = int((b1*c1 + b2*c2) / maxDepth)

  return (r,g,b)

def tree(size, depth=0):
  # konec vƒõtv√≠ - list
  if size<6:
    T.width(randint(2,4))
    T.color(lightgreen)
    size = randint(3,6)
    T.forward(size)
    T.back(size)
    return

  # √∫hel vlevo a vpravo
  al = 31 + randint(-8, 8)
  ar = 29 + randint(-8, 8)
  # velikost podstromu vlevo a vpravo
  sl = .7 + randint(-10, 10) / 100
  sr = .65 + randint(-10, 10) / 100

  # kmen
  T.color(mixColor(depth))
  T.width(int(size)//20 + 1)
  T.forward(size)
  # lev√Ω podstrom
  T.left(al)
  tree(size*sl, depth+1)
  # prav√Ω podstrom
  T.right(al + ar)
  tree(size*sr, depth+1)

  # vycouvat zpƒõt
  T.left(ar)
  T.penup()
  T.back(size)
  T.pendown()

# les√≠k
T.initializeTurtle()
T.shape('circle')
T.speed(13)

for i in range(4):
  T.penup()
  T.goto(120 + 140 * i, 460)
  T.pendown()
  tree(40 + 16 * i)

T.hideturtle()
```
