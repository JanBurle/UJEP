from typing import Any, Optional, TypeAlias


# exception raised by linked list
class LinkedListError(Exception):
  pass


Link: TypeAlias = Optional['LinkedList.Node']  # may be None
Item: TypeAlias = Any  # any type of item


class LinkedList:
  # raise an exception
  @staticmethod
  def _error(msg: str):
    raise LinkedListError(msg)

  @classmethod
  def __TODO(cls, what: str):
    cls._error(f'TODO: {what} not implemented')

  # local class: list node
  class Node:
    def __init__(self, item: Item, link: Link = None):
      self.item = item  # list item
      self.link = link  # link to next node

  # LinkedList constructor
  def __init__(self):
    self._head: Link = None  # first item or None
    self._tail: Link = None  # last item or None
    self._size: int = 0  # item count

  def __len__(self) -> int:  # len(...)
    return self._size

  def isEmpty(self) -> bool:
    return not len(self)

  def append(self, item):
    newNode = self.Node(item)

    if not self._tail:
      # empty list
      self._head = newNode
      self._tail = newNode
    else:
      self._tail.link = newNode
      self._tail = newNode

    self._size += 1

  def prepend(self, item):
    self.__TODO('prepend')

  # ensure that index is valid (optionally +1 for insert)
  def __ensure_valid_index(self, name: str, index: int, plusOne=False):
    if 0 <= index:
      lgt = len(self)
      if index < lgt or (plusOne and index == lgt):
        return  # index is valid
    self._error(f'{name}: bad index {index}')

  # ensure non-empty list
  def __ensure_not_empty(self, name: str):
    if self.isEmpty():
      self._error(f'{name}: empty list')

  def insert(self, index: int, item):
    self.__ensure_valid_index('insert', index, True)

    # as last item
    if index == len(self):
      self.append(item)
      return

    # as first item
    if 0 == index:
      self.prepend(item)
      return

    # anywhere in between
    # index greater than 0 and valid, so the list is not empty
    assert self._head

    # find previous node
    prevNode = self._head
    for _ in range(index - 1):
      assert prevNode.link
      prevNode = prevNode.link

    # insert new node after previous node
    newNode = self.Node(item, prevNode.link)
    prevNode.link = newNode

    self._size += 1

  @property
  # first item
  def front(self) -> Item:
    self.__ensure_not_empty('front')
    assert self._head
    return self._head.item

  @property
  # last item
  def back(self) -> Item:
    self.__ensure_not_empty('back')
    assert self._tail
    return self._tail.item

  # any item
  def get(self, index: int) -> object:
    self.__ensure_valid_index('get', index)

    node = self._head
    for _ in range(index):
      assert node
      node = node.link

def doVal(val):
  print(val, end=' ')

# depth-first - preorder
def traverse_pre(node: list|tuple):
  val, children = node
  doVal(val)
  for child in children:
    traverse_pre(child)
traverse_pre(tree1)

# depth-first - postorder
def traverse_post(node: list|tuple):
  val, children = node
  for child in children:
    traverse_post(child)
  doVal(val)

from collections import deque

def breadthFirst(tree: list|tuple):
  todo = deque([tree])
  while todo:
    node = todo.popleft()
    val, children = node
    doVal(val)
    for child in children:
      todo.append(child)

## Binární strom

```python
from typing import Any, Optional, TypeAlias

BNode: TypeAlias = Optional['BinaryNode']


class BinaryNode:
  @staticmethod
  def __toNode(val: Any) -> BNode:
    if val is None or isinstance(val, BinaryNode):
      return val
    return BinaryNode(val)

  def __init__(self, val: Any, left: Any = None, right: Any = None):
    self.val = val
    self.left = self.__toNode(left)
    self.right = self.__toNode(right)

  def print(self, fill=''):
    print(self.val)
    for i, child in enumerate([self.left, self.right]):
      c1, c2 = ('└─ ', '   ') if 0 < i else ('├─ ', '|  ')
      print(fill + c1, end='')
      if child:
        child.print(fill + c2)
      else:
        print()

  @property
  def prefix(self) -> str:
    res = str(self.val)
    if self.left is not None:
      res += ' ' + self.left.prefix
    if self.right is not None:
      res += ' ' + self.right.prefix
    return res

  @property
  def postfix(self) -> str:
    res = ''
    if self.left is not None:
      res += self.left.postfix + ' '
    if self.right is not None:
      res += self.right.postfix + ' '
    res += str(self.val)
    return res

  @property
  def infix(self) -> str:
    res = ''
    parens = False
    if self.left is not None:
      res += self.left.infix + ' '
      parens = True
    res += str(self.val)
    if self.right is not None:
      res += ' ' + self.right.infix
      parens = True
    return '(' + res + ')' if parens else res


# aliases
N = BinaryNode
BinaryTree = BinaryNode

# tests
N(42, 1, 2).print()

# algebraic expression, binary operators: x = 2*a + 3*(b+c)
bc = N('+', 'b', 'c')
_3bc = N('*', 3, bc)
_2a = N('*', 2, 'a')
expr = BinaryTree('=', 'x', N('+', _2a, _3bc))

expr.print()
print(expr.prefix)
print(expr.postfix)
print(expr.infix)
```

## Uspořádaný binární strom

```python
from typing import Any, Optional, TypeAlias

OBNode: TypeAlias = Optional['OrderedBinaryNode']

# subclass
class OrderedBinaryNode(BinaryNode):
  def __init__(self, val: Any, left: OBNode = None, right: OBNode = None):
    super().__init__(val, left, right)

  def add(self, val):
    if val < self.val:
      if self.left is None:
        self.left = self.__class__(val)
      else:
        self.left.add(val)
    else:
      if self.right is None:
        self.right = self.__class__(val)
      else:
        self.right.add(val)

  def __str__(self):
    res = ''
    if self.left: res += str(self.left)
    res += str(self.val) + ' '
    if self.right: res += str(self.right)
    return res

class OrderedBinaryTree:

  def __init__(self):
    self.root: OBNode = None

  def add(self, val):
    if not self.root:
      self.root = OrderedBinaryNode(val, None, None)
    else:
      self.root.add(val)

  def print(self):
    if self.root is not None:
      self.root.print()

  def __str__(self):
    return '' if self.root is None else str(self.root)

# tests
from random import randint

for _ in range(20): o.add(randint(0,99))

print(o)
o.print()
```
