## Abstraktní datové typy a abstraktní třídy v Pythonu

- **Abstraktní datový typ** lze v Pythonu implementovat pomocí
  _abstraktní třídy_.
- **Abstraktní třída** je třída, která obsahuje alespoň jednu
  _abstraktní metodu_. Abstraktní třída musí dědit z třídy
  `ABC` (Abstract Base Class).
- **Abstraktní metoda** je metoda, která je deklarována, ale
  není implementována. Je označena dekorátorem
  `@abstractmethod` a předpokládá se, že bude implementována v
  podtřídě.

Abstraktní třída nemůže být instanciována (nelze vytvořit
instanci, objekt), ale může být děděna. Podtřída, která dědí z
abstraktní třídy, musí implementovat všechny abstraktní
metody, jinak je také abstraktní třídou.

### Jednoduchý příklad:

Bez abstraktní třídy:

```python
class Zvíře:
  def zvuk(self) -> str:
    return ""

class Pes(Zvíře):
  def zvuk(self) -> str:
    return "Haf!"

class Kočka(Zvíře):
  def zvuk(self) -> str:
    return "Mňau!"

zvířata = [Zvíře(), Pes(), Kočka()]
for z in zvířata:
  print(z.zvuk())
```

S podporou abstraktní třídy:

```python
from abc import ABC, abstractmethod

# abstract base class
class Zvíře(ABC):
  @abstractmethod
  def zvuk(self) -> str:
    pass                           # neimplementovaná metoda

class Pes(Zvíře):
  def zvuk(self) -> str:
    return "Haf!"

class Kočka(Zvíře):
  def zvuk(self) -> str:
    return "Mňau!"

# zvířata = [Zvíře(), Pes(), Kočka()] # TypeError: Cannot instantiate abstract class Zvíře
zvířata = [Pes(), Kočka()] # TypeError: Cannot instantiate abstract class Zvíře
for z in zvířata:
  print(z.zvuk())
```

## Abstraktní orientovaný graf (ADT)

```python
# ABSTRACT DIRECTED GRAPH
# today: without exceptions

from typing import TypeAlias, List, Dict, Tuple
from abc import ABC, abstractmethod

# vertex label
Label: TypeAlias = str

class AbstractGraph(ABC):
  @abstractmethod
  def hasVertex(self, label: Label) -> bool:
    pass

  @abstractmethod
  def addVertex(self, label: Label) -> bool:
    pass

  @abstractmethod
  def hasEdge(self, src: Label, dst: Label) -> bool:
    pass

  @abstractmethod
  def addEdge(self, src: Label, dst: Label) -> bool:
    pass

  @abstractmethod
  def vertexList(self) -> List[Label]:
    pass

  @abstractmethod
  def edgeList(self) -> List[Tuple[Label, Label]]:
    pass

  @abstractmethod
  def edgesFrom(self, src: Label) -> List[Label]:
    pass

  @abstractmethod
  def print(self):
    pass
```

## Graf reprezentovaný jako seznam sousedů (spojová struktura)

```python
# LINKED GRAPH

# concrete implementation of ADT
NodeDict: TypeAlias = Dict[Label, 'LinkedGraph.Node']

class LinkedGraph(AbstractGraph):
  class Node:
    def __init__(self, label: Label):
      self.label = Label
      # links to nodes:
      self.edges: NodeDict = {}

  def __init__(self):
    self.nodes: NodeDict = {}

  def hasVertex(self, label: Label) -> bool:
    return label in self.nodes

  def addVertex(self, label: Label) -> bool:
    if self.hasVertex(label):
      return False # already has such vertex

    self.nodes[label] = self.Node(label)
    return True

  def hasEdge(self, src: Label, dst: Label) -> bool:
    srcNode = self.nodes.get(src)
    return (srcNode is not None) and (dst in srcNode.edges)

  def addEdge(self, src: Label, dst: Label) -> bool:
    srcNode = self.nodes.get(src)
    dstNode = self.nodes.get(dst)

    if srcNode and dstNode and not dst in srcNode.edges:
      srcNode.edges[dst] = dstNode
      return True

    return False

  def vertexList(self) -> List[Label]:
    return list(self.nodes.keys())

  def edgeList(self) -> List[Tuple[Label, Label]]:
    res = []
    for label,node in self.nodes.items():
      for dst in node.edges:
        res.append((label,dst))
    return res

  def edgesFrom(self, label: Label) -> List[Label]:
    node = self.nodes.get(label)
    return list(node.edges.keys()) if node else []

  def print(self):
    for label,node in self.nodes.items():
      edges = [label for label in node.edges]
      print(f"{label}: {edges}")
```

## Graf reprezentovaný jako matice sousedství

```python
# GRAPH, ADJACENCY MATRIX
from typing import List

# concrete implementation of ADT
class AdjGraph(AbstractGraph):
  def __init__(self):
    # vertices: index → label
    self.idxLbl: List[Label] = []
    # vertices: label → index
    self.lblIdx: Dict[Label, int] = {}
    # matrix: index × index
    self.matrix: List[List[bool]] = []

  def hasVertex(self, label: Label) -> bool:
    return label in self.lblIdx

  def addVertex(self, label: Label) -> bool:
    if self.hasVertex(label):
      return False # already has such vertex

    idx = len(self.idxLbl)
    self.idxLbl.append(label)
    self.lblIdx[label] = idx
    for row in self.matrix:
      row.append(False)
    self.matrix.append([False] * (idx+1))
    return True

  def hasEdge(self, src: Label, dst: Label) -> bool:
    srcIdx = self.lblIdx.get(src)
    dstIdx = self.lblIdx.get(dst)

    return \
      srcIdx is not None and \
      dstIdx is not None and \
      self.matrix[srcIdx][dstIdx]

  def addEdge(self, src: Label, dst: Label) -> bool:
    srcIdx = self.lblIdx.get(src)
    dstIdx = self.lblIdx.get(dst)

    if srcIdx is not None and dstIdx is not None \
    and not self.matrix[srcIdx][dstIdx]:
      self.matrix[srcIdx][dstIdx] = True
      return True

    return False

  def edgesFrom(self, label: Label) -> List[Label]:
    srcIdx = self.lblIdx.get(label)
    if srcIdx is None:
      return []
    return [self.idxLbl[idx] \
              for idx,link in enumerate(self.matrix[srcIdx]) if link]

  def vertexList(self) -> List[Label]:
    return self.idxLbl

  def edgeList(self) -> List[Tuple[Label, Label]]:
    res = []
    lbl = self.idxLbl
    lgt = len(lbl)
    for src in range(lgt):
      for dst in range(lgt):
        if self.matrix[src][dst]:
          res.append((lbl[src],lbl[dst]))
    return res

  def print(self):
    lgt = len(self.idxLbl)
    for srcIdx in range(lgt):
      label = self.idxLbl[srcIdx]
      edges = [self.idxLbl[dstIdx] for dstIdx in range(lgt) \
        if self.matrix[srcIdx][dstIdx]]
      print(f"{label}: {edges}")
```

## Hledání cesty v grafu

### do hloubky

```python
# search a path
# https://www.geeksforgeeks.org/difference-between-bfs-and-dfs/

from collections import deque

def dfs_path(g: AbstractGraph, start: Label, goal: Label) -> None|List[Label]:
  stack: List[Tuple[Label, List[Label]]] = [(start, [start])]
  visited: Set[Label] = set()

  while stack:
    (vertex, path) = stack.pop()
    if vertex not in visited:
      if vertex == goal:
        return path
      visited.add(vertex)
      for next in g.edges(vertex):
        stack.append((next, path + [next]))

  return None
```

```python
def bfs_path(graph, start, goal):
```
