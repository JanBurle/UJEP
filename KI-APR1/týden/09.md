Rozdělené do funkcí s ošetřením vstupu:

```python
def vstup() -> tuple[float,float,float]:
  """ Asks user for three floats separated by white space"""
  while True:
    try:
      inp3 = input('Prosím tři čísla oddělená mezerami: ').split()
      a,b,c = map(float, inp3)
      return a,b,c
    except:
      print('Zkuste to znovu.')

def jePravouhly(a:float, b:float, c:float) -> bool:
  """Detects right triangle. C is the hypothenuse."""
  from math import isclose
  return isclose(a**2 + b**2, c**2)

a,b,c = sorted(vstup())
print(jePravouhly(a, b, c))
```

### ❓ Řešitel kvadratických rovnic

Program přijme najednou koeficienty kvadratické rovnice oddělené mezerou. Vypočítá a vypíše kořeny této rovnice.

Prototyp:

```python
def solveQuadraticEquation(a:float, b:float, c:float) -> None|float|tuple[float,float]:
  D = b*b - 4*a*c

  if D < 0:
    return None

  if 0 < D:
    sqrtD = D ** .5
    x1 = (-b + sqrtD) / (2*a)
    x2 = (-b - sqrtD) / (2*a)
    return (x1, x2)

  # here: D==0
  return -b / (2*a)

a,b,c = map(float, input('Zadej a,b,c oddělené mezerou: ').split())
roots = solveQuadraticEquation(a, b, c)
print(f'Kořeny: {roots}')
```

### 🌟 Sazka/sportka

Zadáte šest čísel v rozmezí 1-49. Program vygeneruje šest náhodných čísel a vypíše, kolik čísel jste uhodl/a.

```python
from random import randint

tah = set()
while len(tah) < 6:
  tah.add(randint(1,49))

los = set()
while len(los) < 6:
  try:
    n = int(input("Zadejte číslo mezi 1 a 49: "))
    if 1 <= n <= 49:
      los.add(n)
  except:
    pass

print("Los: ", sorted(los))
print("Tah: ", sorted(tah))

trefa = tah.intersection(los)

print(f"Uhádli jste {sorted(trefa)}, to jest {len(trefa)} správně.")
```

### 😬 Silné heslo

Program zjistí, zda zadané heslo uživatelem je silné.

```python
def isStrong(pwd: str) -> bool:
  if pwd != pwd.replace(' ',''):
    return False # no spaces allowed

  if len(pwd) < 8:
    return False # too short

  hasUpper   = False
  hasLower   = False
  hasDigit   = False
  hasSpecial = False

  for c in pwd:
    if c.isalpha():
      if c==c.upper():
        hasUpper = True
      else:
        hasLower = True
    elif c.isdigit():
      hasDigit = True
    else:
      hasSpecial = True

  return hasUpper and hasLower and hasDigit and hasSpecial

# test

for pwd in pwds:
   print(pwd, isStrong(pwd))
```

### 🌀 Trefa do kulatého terče

Ve čtverci je umístěna kružnice (terč) s daným středem a poloměrem. Program náhodně vystřelí do čtverce a zjistí, zda se trefil.

```python
# souřadnice čtverce
x1 = -10.
x2 = +10.
y1 = -10.
y2 = +10.

# souřadnice kruhu: střed a poloměr
kx = 0.
ky = 0.
kr = 5.

def střela() -> tuple[float,float]:
  from random import uniform
  x = uniform(x1,x2)
  y = uniform(y1,y2)
  return (x,y)

def jeVKruhu(xy: tuple[float,float]) -> bool:
  from math import sqrt

  x,y = xy
  r = sqrt((kx-x)**2 + (ky-y)**2)
  return r < kr

# test
zásah = False
while not zásah:
   pokus = střela()
   zásah = jeVKruhu(pokus)
   print('{:+.1f} {:+.1f}'.format(*pokus), '-', 'ano' if zásah else 'ne')
```

Monte Carlo odhad hodnoty π

```python
x1 = -1.
x2 = +1.
y1 = -1.
y2 = +1.

kx = 0.
ky = 0.
kr = 1.

n = 100_000
c = 0

for _ in range(n):
  if jeVKruhu(střela()):
    c += 1

print(f'pi={4 * c/n}')
```

Rekurzivní verze (složitost O(n)):

```python
def _fib(l:list[int], a:int, b:int, n:int):
  if n<=0: return
  l.append(a)
  _fib(l, b, a+b, n-1)

def fib(n:int)-> list[int]:
  l = []
  _fib(l, 0, 1, n)
  return l

fib(180)
```

### 🖹 Zadaná písmenka

Uživatel postupně zadává různé texty. Vstup skončí jakmile uživatel napíše slovo STOP. Program vypíše jaká písmena se vyskytla v zadaných textech, a kolikrát. Výstup bude seřazený sestupně podle počtu výskytů.

```python
chars = {}

while True:
  # word = input('Slovo: ').strip()
  word = 'Gjdee8  dek'
  if 'STOP'==word:
    break
  for char in word.lower():
    chars[char] = chars.get(char,0) + 1
  break

print(sorted(chars.items(), key=lambda p: p[1], reverse=True))
```
